Const createController

  • createController(path: string, routes: DangoRoute[], middlewares?: DangoMiddleware[]): DangoController
    • Defined in routing/createController.ts:38

    Create a Dango controller with routes, middlewares.

    example
     import { createController } from 'dango-core';

export default createController('/', [
{
path: '/',
method: 'get',
handler: (req, res) => {
res.send('test on home');
},
},
{
path: '/test/:id',
method: 'get',
handler: (req, res) => {
res.send('test with id');
},
middlewares: [
(req, res, next) => {
if (req.params.id !== 'xx') {
next('no param');
}
next();
},
],
},
]);
